# Phase 0 · Foundation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Lay the foundation modules (`app/settings.py`, `app/paths.py`, `app/io_utils.py`) and infrastructure fixes (locked deps, SQLite WAL, atomic JSON writes) that every later refactor phase depends on. **Zero business-logic changes** — only call-site migrations of equivalent semantics.

Migration is incremental: scaffolds first (commits 1–N), then per-file call-site swaps (commits N+1 onward), smoke after each. Pip-tools pins the dep tree so the rest of the refactor can't be silently broken by upstream releases.

**Tech Stack:** pydantic-settings (already installed as FastAPI transitive dep); pip-tools (new dev dep); stdlib `tempfile` for atomic writes; SQLite `PRAGMA journal_mode=WAL`.

```python
"""Atomic JSON write + safe JSON read.

Replaces the 3 places that did `path.write_text(json.dumps(d))` which
left half-written files on crash. Uses tempfile + os.replace for an
atomic swap on the same filesystem (atomic on both POSIX and Windows).

`read_json_safe` returns `default` on missing/corrupt/unreadable file.
"""
from __future__ import annotations

import json
import os
import tempfile
from pathlib import Path
from typing import Any


def write_json_atomic(path: Path | str, data: Any, *, indent: int | None = 2) -> None:
    """Write JSON to *path* atomically.

    Writes to a tempfile in the SAME directory (so os.replace is atomic),
    fsyncs, then renames. Creates parent dirs if missing. On error,
    cleans up the tempfile.
    """
    p = Path(path)
    p.parent.mkdir(parents=True, exist_ok=True)
    fd, tmp_name = tempfile.mkstemp(
        prefix=p.name + ".tmp.",
        dir=str(p.parent),
    )
    try:
        with os.fdopen(fd, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=indent)
            f.flush()
            os.fsync(f.fileno())
        os.replace(tmp_name, p)
    except Exception:
        try:
            os.unlink(tmp_name)
        except OSError:
            pass
        raise


def read_json_safe(path: Path | str, default: Any = None) -> Any:
    """Read JSON from *path*. Return *default* on missing/corrupt/unreadable.

    Catches FileNotFoundError, PermissionError, UnicodeDecodeError,
    JSONDecodeError, OSError. Never raises.
    """
    p = Path(path)
    try:
        text = p.read_text(encoding="utf-8")
    except (FileNotFoundError, PermissionError, UnicodeDecodeError, OSError):
        return default
    try:
        return json.loads(text)
    except json.JSONDecodeError:
        return default
```

```python
"""Centralized application settings.

Single typed source for every environment variable. Imported by both the
phone-bridge main process and the mcp_pb standalone process.

`set -a; . ./.env; set +a`). Also auto-loads `.env` at the repo root for

from pydantic import field_validator
from pydantic_settings import BaseSettings, SettingsConfigDict

from app.paths import BRIDGE_ROOT


class Settings(BaseSettings):
    """All Phone Bridge env vars, typed and defaulted.

    Defaults match the historical `os.environ.get(KEY, DEFAULT)` calls.
    Changing a default == changing user-visible behavior; pin first, refactor later.
    """

    model_config = SettingsConfigDict(
        env_file=str(BRIDGE_ROOT / ".env"),
        env_file_encoding="utf-8",
        extra="ignore",
        case_sensitive=False,

    # PocketBase (shared)
    pocketbase_url: str = ""
    pocketbase_admin_email: str = ""
    pocketbase_admin_password: str = ""
    pb_token: str = ""

    # Bridge runtime
    host: str = "127.0.0.1"

    default_cwd: str = ""
    allowed_origins: str = "*"
    bridge_auth_file: str = ""
    bridge_cookie_days: int = 30
    bridge_name: str = ""
    bridge_data_dir: str = ""

    # Push
    vapid_public_key: str = ""
    vapid_private_key: str = ""

    # POI
    foursquare_key: str = ""
    amap_key: str = ""

    # Notion sync
    notion_token: str = ""
    notion_sync_activity_db_id: str = ""

    # mcp_pb
    mcp_pb_host: str = "127.0.0.1"
    mcp_pb_port: int = 8091
    mcp_pb_public_url: str = ""  # PUBLIC_URL

    @field_validator("pocketbase_url", mode="after")
    @classmethod
    def _strip_trailing_slash(cls, v: str) -> str:
        return v.rstrip("/")


@lru_cache(maxsize=1)
def get_settings() -> Settings:
    """Cached singleton. Tests construct Settings() directly instead."""
    return Settings()


# Module-level convenience.
settings: Settings = get_settings()
```

```python
def _conn():
    conn = sqlite3.connect(DB_PATH, ...)
    conn.row_factory = sqlite3.Row
    # WAL: concurrent readers during writes; survives crash without rollback.
    # NORMAL synchronous is the standard WAL companion.
    # foreign_keys: SQLite default is OFF for back-compat; turn on.
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("PRAGMA synchronous=NORMAL")
    conn.execute("PRAGMA foreign_keys=ON")
    return conn
```
